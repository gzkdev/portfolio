This is a prototype examining how we can use Framer Motion's `layoutId` to create a liquid sliding effect for pill-shaped navigation tabs. The result feels much more organic than a standard cross-fade or immediate jump.

<Sandpack 
  template="react" 
  files={{
    "/App.js": `import { useState } from "react";
import { motion } from "framer-motion";

const tabs = ["Overview", "Integrations", "Activity", "Settings"];

export default function App() {
  const [active, setActive] = useState(tabs[0]);

return (

<div style={{ display: "flex", gap: "8px", padding: "20px", background: "#f5f5f5", borderRadius: "12px" }}>
{tabs.map((tab) => (
<button
key={tab}
onClick={() => setActive(tab)}
style={{
            position: "relative",
            padding: "8px 16px",
            border: "none",
            background: "transparent",
            cursor: "pointer",
            fontWeight: 500,
            fontSize: "14px",
            color: active === tab ? "#000" : "#666",
            zIndex: 1
          }} >
{active === tab && (
<motion.div
layoutId="bubble"
style={{
                position: "absolute",
                inset: 0,
                backgroundColor: "white",
                borderRadius: "99px",
                boxShadow: "0 2px 5px rgba(0,0,0,0.1)",
                zIndex: -1
              }}
transition={{ type: "spring", bounce: 0.2, duration: 0.6 }}
/>
)}
{tab}
</button>
))}
</div>
);
}`
}}
/>
